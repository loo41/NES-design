<Playground wrapper="CheckboxGroup / Option is Object">
  {
    () => {
      const [value, setValue] = useState([])
      return (
        <>
          <CheckboxGroup
            value={value}
            options={[
              { label: 'Yes', value: 'yes', defaultChecked: true },
              { label: 'No', value: 'no'},
            ]}
            onChange={values => setValue(values)}
          />
          <CheckboxGroup
            theme="dark"
            style={{
              display: 'flex',
              background: 'black', 
              height: 50, 
              alignItems: 'center'
            }}
            value={value}
            options={[
              { label: 'Yes', value: 'yes', defaultChecked: true },
              { label: 'No', value: 'no'},
            ]}
            onChange={values => setValue(values)}
          />
        </>
      )
    }
  }
</Playground>

<Playground wrapper="CheckboxGroup + CheckBox">
  {
    () => {
      const [value, setValue] = useState(['orange'])
      return (
        <CheckboxGroup
          value={value}
          options={[
            { label: 'Apple', value: 'apple', defaultChecked: true },
            { label: 'Banana', value: 'banana'},
          ]}
          onChange={values => setValue(values)}
        >
          <CheckBox labelProps={{style: { display: 'block' }}} value="orange" children="Orange" />
        </CheckboxGroup>
      )
    }
  }
</Playground>
